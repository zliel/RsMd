3/// Utility function for pushing a String buffer to a generic collection.
4///
5/// This function checks if the buffer is not empty, converts it to the type `T`, and pushes it to
6/// the collection.
7///
8/// # Type Parameters
9/// * `T` - The type to which the buffer will be converted. It must implement the
10///   `From<String>` trait.
11///
12/// # Arguments
13/// * `collection` - A mutable reference to the collection of type `Vec<T>` where the buffer will be pushed.
14/// * `buffer` - A mutable reference to the String buffer that will be converted and pushed.
15pub fn push_buffer_to_collection<T>(collection: &mut Vec<T>, buffer: &mut String)
16where
17    T: From<String>,
18{
19    if !buffer.is_empty() {
20        collection.push(T::from(buffer.to_string()));
21        buffer.clear();
22    }
23}

25/// Builds a relative prefix path based on the provided HTML relative path.
26/// Note that this function does not add the final file name, only the preceding directories.
27pub fn build_rel_prefix(html_rel_path: &str) -> PathBuf {
28    let rel_path = Path::new(html_rel_path);
29    let depth = rel_path.parent().map_or(0, |p| p.components().count());
30    let mut rel_prefix = PathBuf::new();
31    for _ in 0..depth {
32        rel_prefix.push("..");
33    }
34    rel_prefix
35}
