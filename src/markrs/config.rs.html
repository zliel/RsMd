79impl Config {
80    /// Creates a new `Config` instance from the specified file path
81    ///
82    /// # Arguments
83    /// * `file_path` - The path to the configuration file. If no file path is provided as a CLI
84    ///   arg, it will check for a config file in the default config directory.
85    ///
86    /// # Returns
87    /// Returns a `Result` containing the `Config` instance if successful
88    pub fn from_file(file_path: &str) -> Result<Self, String> {
89        // If the user provided a config file, try to load the config from it
90        if !file_path.is_empty() {
91            info!("Loading config from file: {}", file_path);
92            let contents = std::fs::read_to_string(file_path)
93                .map_err(|e| format!("Failed to read config file: {}", e))?;
94
95            let config: Config = toml_edit::de::from_str(&contents)
96                .map_err(|e| format!("Failed to parse config file: {}", e))?;
97
98            validate_config(file_path, &contents, &config)?;
99
100            return Ok(config);
101        }
102
103        let config_path =
104            get_config_path().map_err(|e| format!("Failed to get config path: {}", e))?;
105
106        // If the user did not provide a config file, check if a config file exists in the config
107        // directory
108        if does_config_exist()? {
109            let contents = std::fs::read_to_string(&config_path)
110                .map_err(|e| format!("Failed to read config file: {}", e))?;
111
112            let config: Config = toml_edit::de::from_str(&contents)
113                .map_err(|e| format!("Failed to parse config file: {}", e))?;
114
115            validate_config(&config_path.to_string_lossy(), &contents, &config)?;
116
117            Ok(config)
118        } else {
119            warn!(
120                "No config file found, writing default config to: {}",
121                config_path.to_string_lossy()
122            );
123
124            let default_config = write_default_config()
125                .map_err(|e| format!("Failed to write default config: {}", e))?;
126
127            Ok(default_config)
128        }
129    }
130}

132/// Validates the configuration by checking if the original config file matches the filled config
133///
134/// If the original config is missing fields, it updates the file with any missing fields
135fn validate_config(file_path: &str, contents: &str, config: &Config) -> Result<(), String> {
136    let mut doc = toml_edit::DocumentMut::from_str(contents)
137        .map_err(|e| format!("Failed to create TOML document: {}", e))?;
138
139    let filled_doc = toml_edit::ser::to_document(config)
140        .map_err(|e| format!("Failed to serialize config to TOML: {}", e))?;
141
142    let mut config_needs_update = false;
143    let mut missing_fields = Vec::new();
144    for (section, values) in filled_doc.iter() {
145        let table = values.clone().into_table().unwrap_or_else(|_item| {
146            error!(
147                "Expected a table for field '{}', but found: {}",
148                section, values
149            );
150            panic!("Invalid configuration format for field '{}'", section);
151        });
152
153        for (sub_key, sub_value) in table.iter() {
154            if !doc.contains_key(section) {
155                doc[section] = filled_doc[section].clone();
156                config_needs_update = true;
157                missing_fields.push(section.to_string());
158            } else if !doc[section].is_table()
159                || !doc[section].as_table().unwrap().contains_key(sub_key)
160            {
161                doc[section][sub_key] = sub_value.clone();
162                config_needs_update = true;
163                missing_fields.push(format!("{}.{}", section, sub_key));
164            }
165        }
166    }
167
168    if config_needs_update {
169        warn!(
170            "Config is missing fields: {:?}, writing updated config to: {}",
171            missing_fields, file_path
172        );
173
174        // Formats the file with sections like `[lexer]` and `tab_size = 4`
175        // previously it would be `lexer = { tab_size = 4 }`
176        if !doc["lexer"].is_table() {
177            doc["lexer"] = doc["lexer"]
178                .clone()
179                .into_table()
180                .unwrap_or_else(|_item| {
181                    error!(
182                        "Expected 'lexer' to be a table, but found: {}",
183                        doc["lexer"]
184                    );
185                    panic!("Invalid configuration format for 'lexer'");
186                })
187                .into();
188        }
189        doc["lexer"].as_table_mut().unwrap().set_position(0);
190
191        if !doc["html"].is_table() {
192            doc["html"] = doc["html"]
193                .clone()
194                .into_table()
195                .unwrap_or_else(|_item| {
196                    error!("Expected 'html' to be a table, but found: {}", doc["html"]);
197                    panic!("Invalid configuration format for 'html'");
198                })
199                .into();
200        }
201        doc["html"].as_table_mut().unwrap().sort_values();
202
203        std::fs::write(file_path, doc.to_string())
204            .map_err(|e| format!("Failed to write config file: {}", e))?;
205    }
206
207    Ok(())
208}

210/// Initializes the global configuration from the specified file path
211///
212/// # Arguments
213/// * `config_path` - The path to the configuration file.
214///
215/// # Returns
216/// Returns a `Result` indicating success or failure. If successful, a global `CONFIG` has been
217/// initialized.
218pub fn init_config(config_path: &str) -> Result<(), Box<dyn std::error::Error>> {
219    CONFIG.get_or_init(|| {
220        Config::from_file(config_path).unwrap_or_else(|err| {
221            error!("Failed to load config: {}", err);
222            std::process::exit(1);
223        })
224    });
225    Ok(())
226}
