9/// Generates an HTML string from a vector of MdBlockElements
10///
11/// # Arguments
12/// * `file_name` - The name of the markdown file, used to set the title of the HTML document.
13/// * `md_elements` - A vector of `MdBlockElement` instances representing the markdown content.
14/// * `output_dir` - The directory where the generated HTML file will be saved.
15/// * `input_dir` - The directory where the markdown files are located, used for relative paths.
16/// * `html_rel_path` - The relative path to the HTML file from the output directory, used for
17///   linking resources.
18///
19/// # Returns
20/// Returns a `String` containing the generated HTML.
21pub fn generate_html(
22    file_name: &str,
23    md_elements: &[MdBlockElement],
24    output_dir: &str,
25    input_dir: &str,
26    html_rel_path: &str,
27) -> String {
28    let mut html_output = String::new();
29
30    let head = generate_head(file_name, html_rel_path);
31
32    let mut body = String::from("\t<body>\n");
33    body.push_str(&indent_html(&generate_navbar(html_rel_path), 2));
34    body.push_str("\n\t\t<div id=\"content\">");
35
36    let inner_html: String = md_elements
37        .iter()
38        .map(|element| element.to_html(output_dir, input_dir, html_rel_path))
39        .collect::<Vec<String>>()
40        .join("\n");
41
42    let inner_html = if CONFIG.get().unwrap().html.sanitize_html {
43        ammonia::Builder::default()
44            .add_tag_attributes("a", &["href", "title", "target"])
45            .add_tag_attribute_values("a", "target", &["_blank", "_self"])
46            .add_tag_attributes("pre", &["class"])
47            .add_tag_attributes("code", &["class"])
48            .add_tags(&["iframe"])
49            .add_tag_attributes(
50                "iframe",
51                &[
52                    "src",
53                    "width",
54                    "height",
55                    "title",
56                    "frameborder",
57                    "allowfullscreen",
58                ],
59            )
60            .clean(&inner_html)
61            .to_string()
62    } else {
63        inner_html
64    };
65
66    body.push_str(&indent_html(&inner_html, 3));
67    body.push_str("\n\t\t</div>");
68
69    if CONFIG.get().unwrap().html.use_prism {
70        body.push_str(

74        body.push_str(
75            "\n\t\t<script src=\"https://cdnjs.cloudflare.com/ajax/libs/prism/1.30.0/plugins/autoloader/prism-autoloader.min.js\" integrity=\"sha512-SkmBfuA2hqjzEVpmnMt/LINrjop3GKWqsuLSSB3e7iBmYK7JuWw4ldmmxwD9mdm2IRTTi0OxSAfEGvgEi0i2Kw==\" crossorigin=\"anonymous\" referrerpolicy=\"no-referrer\"></script>"

91/// Generates the index HTML file that lists all pages
92///
93/// # Arguments
94/// * `file_names` - A slice of `String` containing the names of the markdown files.
95///
96/// # Returns
97/// Returns a `String` containing the generated HTML for the index page.
98pub fn generate_index(file_names: &[String]) -> String {
99    let mut html_output = String::new();
100
101    let head = generate_head("index", "index.html");
102
103    let mut body = String::from("\t<body>\n");
104    body.push_str(&generate_navbar("index.html"));
105    body.push_str("\n\t<div id=\"content\">\n");
106    body.push_str("<h1>All Pages</h1>\n");
107
108    file_names.iter().for_each(|file_name| {
109        body.push_str(&format!(
110            "<a href=\"./{}.html\">{}</a><br>\n",
111            file_name.trim_end_matches(".md"),
112            format_title(file_name)
113        ));
114    });
115
116    body.push_str("\n</div>\n\t</body>\n");
117
118    html_output.push_str(&head);
119    html_output.push_str(&body);
120    html_output.push_str("</html>\n");
121
122    html_output
123}

125/// Generates the HTML head section
126///
127/// # Arguments
128/// * `file_name` - The name of the markdown file, used to set the title of the HTML document.
129/// * `html_rel_path` - The relative path to the HTML file from the output directory, used for
130///   linking
131fn generate_head(file_name: &str, html_rel_path: &str) -> String {
132    let config = CONFIG.get().unwrap();
133    let mut head = String::from(
134        r#"<!DOCTYPE html>
135    <html lang="en">
136    <head>
137        <meta charset="UTF-8">
138        <meta name="viewport" content="width=device-width, initial-scale=1.0">
139    "#,
140    );
141
142    // Remove the file extension from the file name and make it title case
143    let title = format_title(file_name);
144    head.push_str(&format!("\t<title>{}</title>\n", title));
145
146    let favicon_file = config.html.favicon_file.clone();
147    if !favicon_file.is_empty() {
148        let mut favicon_path = build_rel_prefix(html_rel_path);
149        favicon_path.push("media");
150        favicon_path.push(favicon_file.rsplit("/").next().unwrap());
151        let favicon_href = favicon_path.to_string_lossy();
152
153        head.push_str(&format!(
154            "\t<link rel=\"icon\" href=\"{}\">\n",
155            favicon_href
156        ));
157    }
158
159    let css_file = config.html.css_file.clone();
160    let mut css_path = build_rel_prefix(html_rel_path);
161    css_path.push("styles.css");
162    let css_href = css_path.to_string_lossy();
163
164    if css_file == "default" {
165        head.push_str(&format!(
166            "\t\t<link rel=\"stylesheet\" href=\"{}\">\n",
167            css_href
168        ));
169    } else {
170        head.push_str(&format!(
171            "\t\t<link rel=\"stylesheet\" href=\"{}\">\n",
172            css_file
173        ));
174    }
175
176    if config.html.use_prism {
177        if !config.html.prism_theme.is_empty() {
178            let theme = if config.html.sanitize_html {
179                clean(&config.html.prism_theme)
180            } else {
181                config.html.prism_theme.clone()
182            };
183
184            head.push_str(&format!("\t\t<link rel=\"stylesheet\" href=\"https://cdnjs.cloudflare.com/ajax/libs/prism-themes/1.9.0/prism-{}.min.css\" crossorigin=\"anonymous\" referrerpolicy=\"no-referrer\" />", theme));
185        } else {
186            head.push_str("\t\t<link rel=\"stylesheet\" href=\"https://cdn.jsdelivr.net/npm/prismjs@1.30.0/themes/prism-okaidia.min.css\">");
187        }
188        head.push_str("\t\t<link rel=\"stylesheet\" href=\"https://cdnjs.cloudflare.com/ajax/libs/prism/1.30.0/plugins/toolbar/prism-toolbar.min.css\" integrity=\"sha512-Dqf5696xtofgH089BgZJo2lSWTvev4GFo+gA2o4GullFY65rzQVQLQVlzLvYwTo0Bb2Gpb6IqwxYWtoMonfdhQ==\" crossorigin=\"anonymous\" referrerpolicy=\"no-referrer\" />");
189        head.push_str("\t\t<link rel=\"stylesheet\" href=\"https://cdnjs.cloudflare.com/ajax/libs/prism/1.30.0/plugins/line-numbers/prism-line-numbers.min.css\" integrity=\"sha512-cbQXwDFK7lj2Fqfkuxbo5iD1dSbLlJGXGpfTDqbggqjHJeyzx88I3rfwjS38WJag/ihH7lzuGlGHpDBymLirZQ==\" crossorigin=\"anonymous\" referrerpolicy=\"no-referrer\" />");
190    }
191
192    head.push_str("\t</head>\n");
193    head
194}

196/// Generates the HTML for the navigation bar
197fn generate_navbar(html_rel_path: &str) -> String {
198    let mut navbar = String::from("<header>\n\t<nav>\n\t\t<ul>\n");
199
200    let mut home_path = build_rel_prefix(html_rel_path);
201    home_path.push("index.html");
202    let home_href = home_path.to_string_lossy();
203
204    navbar.push_str(&format!(
205        "\t\t\t<li><a href=\"{}\">Home</a></li>",
206        home_href
207    ));
208    navbar.push_str("\n\t\t</ul>\n\t</nav>\n</header>\n\n");
209    navbar
210}
211/// Formats the file name to create a title for the HTML document
212///
213/// # Arguments
214/// * `file_name` - The name of the file, typically ending with `.md`.
215///
216/// # Returns
217/// The formatted title (i.e. "my_test_page.md" -> "My Test Page")
218fn format_title(file_name: &str) -> String {
219    let title = file_name.trim_end_matches(".md").replace('_', " ");
220
221    title
222        .split_whitespace()
223        .map(|word| {
224            let mut chars = word.chars();
225            match chars.next() {
226                Some(first) => first.to_uppercase().collect::<String>() + chars.as_str(),
227                None => String::new(),
228            }
229        })
230        .collect::<Vec<String>>()
231        .join(" ")
232}

234/// Indents each line of the given HTML string by the specified number of tabs.
235pub fn indent_html(html: &str, level: usize) -> String {
236    let indent = "\t".repeat(level);
237    html.lines()
238        .map(|line| {
239            let first_non_whitespace_token = line.chars().find(|c| !c.is_whitespace());
240
241            match first_non_whitespace_token {
242                Some('<') => format!("{indent}{line}"),
243                Some(_) => line.to_string(),
244                None => line.to_string(), // If the line is empty or only whitespace, return it unchanged
245            }
246        })
247        .collect::<Vec<_>>()
248        .join("\n")
249}
