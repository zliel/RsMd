9pub trait ToHtml {

14/// Represents the different types of tokens that can be found in a markdown line.
15#[derive(Debug, PartialEq, Clone)]
16pub enum Token {
17    Text(String),
18    EmphasisRun { delimiter: char, length: usize },
19    Punctuation(String),
20    OpenBracket,
21    CloseBracket,
22    OpenParenthesis,
23    CloseParenthesis,
24    TableCellSeparator,
25    OrderedListMarker(String),
26    Whitespace,
27    CodeTick,
28    CodeFence,
29    ThematicBreak,
30    Escape(String),
31    Tab,
32    Newline,
33    BlockQuoteMarker,
34    RawHtmlTag(String),
35}
36
37impl From<String> for Token {
38    fn from(s: String) -> Self {
39        Token::Text(s.to_string())
40    }
41}

43/// Represents block-level markdown elements.
44#[derive(Debug, PartialEq)]
45pub enum MdBlockElement {
46    Header {
47        level: u8,
48        content: Vec<MdInlineElement>,
49    },
50    Paragraph {
51        content: Vec<MdInlineElement>,
52    },
53    CodeBlock {
54        language: Option<String>,
55        lines: Vec<String>,
56    },
57    ThematicBreak,
58    UnorderedList {
59        items: Vec<MdListItem>,
60    },
61    OrderedList {
62        items: Vec<MdListItem>,
63    },
64    Table {
65        headers: Vec<MdTableCell>,
66        body: Vec<Vec<MdTableCell>>,
67    },
68    BlockQuote {
69        content: Vec<MdBlockElement>,
70    },
71    RawHtml {
72        content: String,
73    },
74}
75
76impl ToHtml for MdBlockElement {
77    fn to_html(&self, output_dir: &str, input_dir: &str, html_rel_path: &str) -> String {
78        match self {
79            MdBlockElement::Header { level, content } => {
80                let inner_html = content
81                    .iter()
82                    .map(|el| el.to_html(output_dir, input_dir, html_rel_path))
83                    .collect::<String>();
84
85                format!("\n<h{level}>{inner_html}</h{level}>\n")
86            }
87            MdBlockElement::Paragraph { content } => {
88                let inner_html = content
89                    .iter()
90                    .map(|el| el.to_html(output_dir, input_dir, html_rel_path))
91                    .collect::<String>();
92                format!("<p>{inner_html}</p>")
93            }
94            MdBlockElement::CodeBlock { language, lines } => {
95                let language_class = match language {
96                    Some(language) => format!("language-{language}"),
97                    None => "language-none".to_string(),
98                };
99
100                if CONFIG.get().unwrap().html.use_prism {
101                    let code = lines.join("\n");
102
103                    format!(
104                        "<pre class=\"{language_class} line-numbers\" style=\"white-space: pre-wrap;\" data-prismjs-copy=\"ðŸ“‹\">\n<code class=\"{language_class} line-numbers\">{code}</code></pre>"
105                    )
106                } else {
107                    let code = lines
108                        .iter()
109                        .map(|line| format!("<code class=\"non_prism\">{line}</code>"))
110                        .collect::<String>();
111
112                    format!("<pre class=\"non_prism\">{code}</pre>")
113                }
114            }
115            MdBlockElement::ThematicBreak => "<hr>".to_string(),
116            MdBlockElement::UnorderedList { items } => {
117                let inner_items = items
118                    .iter()
119                    .map(|item| item.to_html(output_dir, input_dir, html_rel_path))
120                    .collect::<String>();
121
122                let inner_items = indent_html(&inner_items, 1);
123                format!("<ul>\n{inner_items}\n</ul>")
124            }
125            MdBlockElement::OrderedList { items } => {
126                let inner_items = items
127                    .iter()
128                    .map(|item| item.to_html(output_dir, input_dir, html_rel_path))
129                    .collect::<String>();
130
131                let inner_items = indent_html(&inner_items, 1);
132                format!("<ol>\n{inner_items}\n</ol>")
133            }
134            MdBlockElement::Table { headers, body } => {
135                let header_html = headers
136                    .iter()
137                    .map(|cell| cell.to_html(output_dir, input_dir, html_rel_path))
138                    .collect::<Vec<_>>()
139                    .join("\n");
140
141                let header_html = indent_html(&header_html, 3);
142
143                let body_html = body
144                    .iter()
145                    .map(|row| {
146                        let cell_html = row
147                            .iter()
148                            .map(|cell| cell.to_html(output_dir, input_dir, html_rel_path))
149                            .collect::<Vec<_>>()
150                            .join("\n");
151
152                        let cell_html = indent_html(&cell_html, 1);
153
154                        format!("<tr>\n{cell_html}\n</tr>")
155                    })
156                    .collect::<Vec<_>>()
157                    .join("\n");
158
159                let body_html = indent_html(&body_html, 2);
160
161                format!(
162                    "<table>\n\t<thead>\n\t\t<tr>\n{header_html}\n\t\t</tr>\n\t</thead>\n\t<tbody>\n{body_html}\n\t</tbody>\n</table>"
163                )
164            }
165            MdBlockElement::BlockQuote { content } => {
166                let inner_html = content
167                    .iter()
168                    .map(|el| el.to_html(output_dir, input_dir, html_rel_path))
169                    .collect::<String>();
170
171                format!("<blockquote>\n{inner_html}\n</blockquote>")
172            }
173            MdBlockElement::RawHtml { content } => {
174                format!("{}\n", content)
175            }
176        }
177    }
178}

180/// Represents a list item in markdown, which can contain block elements.
181///
182/// # Fields
183/// * `content` - The content of the list item, which can be any block-level markdown element.
184#[derive(Debug, PartialEq)]
185pub struct MdListItem {
186    pub content: MdBlockElement,
187}
188
189impl ToHtml for MdListItem {
190    fn to_html(&self, output_dir: &str, input_dir: &str, html_rel_path: &str) -> String {
191        match &self.content {
192            MdBlockElement::UnorderedList { items } => {
193                let inner_items = items
194                    .iter()
195                    .map(|item| item.to_html(output_dir, input_dir, html_rel_path))
196                    .collect::<String>();
197                let inner_items = indent_html(&inner_items, 1);
198                format!("<ul>\n{inner_items}\n</ul>")
199            }
200            MdBlockElement::OrderedList { items } => {
201                let inner_items = items
202                    .iter()
203                    .map(|item| item.to_html(output_dir, input_dir, html_rel_path))
204                    .collect::<String>();
205                format!("<ol>\n{inner_items}\n</ol>")
206            }
207            _ => {
208                let inner_html = indent_html(
209                    &self.content.to_html(output_dir, input_dir, html_rel_path),
210                    1,
211                );
212                format!("<li>\n{inner_html}\n</li>\n")
213            }
214        }
215    }
216}

218/// Represents a cell in a markdown table.
219#[derive(Debug, PartialEq, Clone)]
220pub struct MdTableCell {
221    pub content: Vec<MdInlineElement>,
222    pub alignment: TableAlignment,
223    pub is_header: bool,
224}
225
226impl ToHtml for MdTableCell {
227    fn to_html(&self, output_dir: &str, input_dir: &str, html_rel_path: &str) -> String {
228        let inner_html = self
229            .content
230            .iter()
231            .map(|el| el.to_html(output_dir, input_dir, html_rel_path))
232            .collect::<String>();
233
234        let text_alignment = match self.alignment {
235            TableAlignment::Left | TableAlignment::None => "left",
236            TableAlignment::Center => "center",
237            TableAlignment::Right => "right",
238        };
239
240        match self.is_header {
241            true => format!("<th style=\"text-align:{text_alignment};\">{inner_html}</th>"),
242            false => format!("<td style=\"text-align:{text_alignment};\">{inner_html}</td>"),
243        }
244    }
245}

256/// Represents inline markdown elements (text, bold/italic, link, etc.)
257#[derive(Debug, PartialEq, Clone)]
258pub enum MdInlineElement {
259    Text {
260        content: String,
261    },
262    Bold {
263        content: Vec<MdInlineElement>,
264    },
265    Italic {
266        content: Vec<MdInlineElement>,
267    },
268    Link {
269        text: Vec<MdInlineElement>,
270        title: Option<String>,
271        url: String,
272    },
273    Image {
274        alt_text: String,
275        title: Option<String>,
276        url: String,
277    },
278    Code {
279        content: String,
280    },
281    Placeholder,
282}

288        }
289    }

292impl ToHtml for MdInlineElement {
293    fn to_html(&self, output_dir: &str, input_dir: &str, html_rel_path: &str) -> String {
294        match self {
295            MdInlineElement::Text { content } => content.clone(),
296            MdInlineElement::Bold { content } => {
297                let inner_html = content
298                    .iter()
299                    .map(|el| el.to_html(output_dir, input_dir, html_rel_path))
300                    .collect::<String>();
301                format!("<b>{}</b>", inner_html)
302            }
303            MdInlineElement::Italic { content } => {
304                let inner_html = content
305                    .iter()
306                    .map(|el| el.to_html(output_dir, input_dir, html_rel_path))
307                    .collect::<String>();
308                format!("<i>{}</i>", inner_html)
309            }
310            MdInlineElement::Link { text, title, url } => {
311                let label_html = text
312                    .iter()
313                    .map(|el| el.to_html(output_dir, input_dir, html_rel_path))
314                    .collect::<String>();
315
316                if url.contains("youtube.com") && url.contains("v=") {
317                    let video_id = url
318                        .split("v=")
319                        .nth(1)
320                        .and_then(|s| s.split('&').next())
321                        .unwrap_or("");
322
323                    return format!(
324                        r#"<div class="video-container">
325                        <iframe width="560" height="315" src="https://www.youtube.com/embed/{}" 
326                        title="YouTube video player" frameborder="0" allowfullscreen></iframe>
327                        </div>"#,
328                        video_id
329                    );
330                }
331
332                // Links to external URLs will open in a new tab
333                if url.starts_with("http") {
334                    match title {
335                        Some(text) => {
336                            format!(
337                                "<a href=\"{url}\" title=\"{text}\" target=\"_blank\">{label_html}â®º</a>"
338                            )
339                        }
340                        None => format!("<a href=\"{url}\" target=\"_blank\">{label_html}â®º</a>"),
341                    }
342                } else {
343                    match title {
344                        Some(text) => {
345                            format!("<a href=\"{url}\" title=\"{text}\">{label_html}</a>")
346                        }
347                        None => format!("<a href=\"{url}\">{label_html}</a>"),
348                    }
349                }
350            }
351            MdInlineElement::Image {
352                alt_text,
353                title,
354                url,
355            } => {
356                let mut media_url = url.clone();
357
358                // If the image uses a relative path, copy it to the output directory
359                if !url.starts_with("http") {
360                    if let Err(e) = copy_image_to_output_dir(url, output_dir, input_dir) {
361                        warn!("Unable to copy image {url}: {e}");
362                    }
363
364                    // Update the URL to point to the copied image in the output directory
365                    let url = url.rsplit('/').next().unwrap_or(url);
366
367                    let rel_prefix = build_rel_prefix(html_rel_path);
368
369                    media_url = format!("./{}/media/{}", rel_prefix.to_string_lossy(), url);
370                }
371
372                match title {
373                    Some(text) => {
374                        format!("<img src=\"{media_url}\" alt=\"{alt_text}\" title=\"{text}\"/>")
375                    }
376                    None => format!("<img src=\"{media_url}\" alt=\"{alt_text}\"/>"),
377                }
378            }
379            MdInlineElement::Code { content } => format!("<code>{content}</code>"),
380            MdInlineElement::Placeholder => unreachable!(),
381        }
382    }
383}

385/// Cursor for navigating through a vector of tokens
386///
387/// This struct provides methods to access the current token, peek ahead or behind, and advance the
388/// cursor position.
389///
390/// # Fields
391/// * `tokens` - A vector of tokens to navigate through.
392/// * `current_position` - The current position of the cursor within the token vector.
393#[derive(Debug)]
394pub struct TokenCursor {
395    pub tokens: Vec<Token>,
396    pub current_position: usize,
397}
398
399impl TokenCursor {
400    /// Returns the current token, if any.
401    pub fn current(&self) -> Option<&Token> {
402        self.tokens.get(self.current_position)
403    }
404
405    /// Returns the nth next token, if any.
406    ///
407    /// # Arguments
408    /// * `n` - The number of tokens to look ahead.
409    ///
410    /// # Returns
411    /// An `Option` containing a reference to the token if it exists, or `None` if it is out of
412    /// bounds.
413    pub fn peek_ahead(&self, n: usize) -> Option<&Token> {
414        self.tokens.get(self.current_position + n)
415    }
416
417    /// Returns the nth previous token, if any.
418    ///
419    /// # Arguments
420    /// * `n` - The number of tokens to look behind.
421    ///
422    /// # Returns
423    /// An `Option` containing a reference to the token if it exists, or `None` if it is out of
424    pub fn _peek_behind(&self, n: usize) -> Option<&Token> {
425        self.tokens.get(self.current_position - n)
426    }
427
428    /// Moves the cursor forward one position.
429    pub fn advance(&mut self) {
430        if self.current_position < self.tokens.len() {
431            self.current_position += 1;
432        }
433    }
434
435    /// Sets the cursor's position to the specified position.
436    ///
437    /// # Arguments
438    /// * `pos` - The position to set the cursor to.
439    ///
440    /// # Panics
441    /// Panics if the position is out of bounds for the token list.
442    pub fn _set_position(&mut self, pos: usize) {
443        if pos < self.tokens.len() {
444            self.current_position = pos;
445        } else {
446            panic!("Position {pos} is out of bounds for the TokenCursor");
447        }
448    }
449
450    /// Returns the current position of the cursor.
451    pub fn position(&self) -> usize {
452        self.current_position
453    }
454
455    /// Returns whether the cursor is at the end of the token list.
456    pub fn is_at_eof(&self) -> bool {
457        self.current_position >= self.tokens.len()
458    }
459}

461/// Manages Delimiter runs in a markdown document.
462/// A delimiter run is a sequence of the same character (e.g., `*`, `_`, `~`) that can be used for
463/// bold/italic writing.
464///
465/// # Fields
466/// * `ch` - The character that represents the delimiter (e.g., `*`, `_`, `~`).
467/// * `run_length` - The number of times the delimiter character appears in a row.
468/// * `token_position` - The position of the first token in this delimiter run.
469/// * `parsed_position` - The position in the `Vec<MdInlineElement>` where the content of this
470///   delimiter run will be stored.
471/// * `active` - Whether this delimiter run is currently active (i.e., it has not been closed).
472/// * `can_open` - Whether this delimiter can open a new emphasis run (e.g., it is left-flanking).
473/// * `can_close` - Whether this delimiter can close an existing emphasis run (e.g., it is
474///   right-flanking).
475#[derive(Debug, Clone)]
476pub struct Delimiter {
477    pub ch: char,
478    pub run_length: usize,
479    pub token_position: usize,
480    pub parsed_position: usize,
481    pub active: bool,
482    pub can_open: bool,  //Must be left-flanking
483    pub can_close: bool, //Must be right-flanking
484}
485
486impl Delimiter {
487    /// Determines whether a delimiter is "Left", "Right", or "Both" flanking
488    /// For exmample, it is left flanking if it's not followed by non-whitespace, and either:
489    /// 1. Not followed by punctuation
490    /// 2. Followed by punctuation and preceded by whitespace or punctuation
491    ///
492    /// Modifies the `can_open` and `can_close` fields in-place based on the classification.
493    ///
494    /// See i<https://spec.commonmark.org/0.31.2/#left-flanking-delimiter-run> for more information.
495    ///
496    /// # Arguments
497    /// * `tokens` - A slice of tokens to classify the delimiter against.
498    pub fn classify_flanking(&mut self, tokens: &[Token]) {
499        let before = if self.token_position > 0 {
500            Some(&tokens[self.token_position - 1])
501        } else {
502            None
503        };
504
505        let after = tokens.get(self.token_position + 1);
506        let followed_by_whitespace = after.is_none_or(is_whitespace);
507        let followed_by_punctuation = after.is_some_and(is_punctuation);
508
509        let preceded_by_whitespace = before.is_none_or(is_whitespace);
510        let preceded_by_punctuation = before.is_some_and(is_punctuation);
511
512        let is_left_flanking = if followed_by_whitespace {
513            false
514        } else if !followed_by_punctuation {
515            true
516        } else {
517            preceded_by_whitespace || preceded_by_punctuation
518        };
519
520        let is_right_flanking = if preceded_by_whitespace {
521            false
522        } else if !preceded_by_punctuation {
523            true
524        } else {
525            followed_by_whitespace || followed_by_punctuation
526        };
527
528        let delimiter_char = self.ch;
529
530        // Apply Rule of 3 and underscore restrictions
531        let is_underscore = delimiter_char == '_';
532
533        if is_underscore {
534            self.can_open = is_left_flanking && (!is_right_flanking || followed_by_punctuation);
535
536            self.can_close = is_right_flanking && (!is_left_flanking || followed_by_punctuation);
537        } else {
538            self.can_open = is_left_flanking;
539            self.can_close = is_right_flanking;
540        }
541    }
542}

544/// Helper function to determine if a token is whitespace or newline.
545///
546/// # Arguments
547/// * `token` - The token to check.
548fn is_whitespace(token: &Token) -> bool {
549    matches!(token, Token::Newline | Token::Whitespace)
550}
551
552/// Helper function to determine if a token is punctuation.
553///
554/// # Arguments
555/// * `token` - The token to check.
556fn is_punctuation(token: &Token) -> bool {
557    matches!(
558        token,
559        Token::Punctuation(_)
560            | Token::EmphasisRun {
561                delimiter: _,
562                length: _
563            }
564            | Token::OpenBracket
565            | Token::CloseBracket
566            | Token::OpenParenthesis
567            | Token::CloseParenthesis
568    )
