16/// Parses a vector of tokenized markdown lines into a vector of block-level Markdown elements.
17///
18/// # Arguments
19/// * `markdown_lines` - A vector of vectors, where each inner vector contains tokens representing a line of markdown.
20///
21/// # Returns
22/// A vector of parsed block-level Markdown elements.
23pub fn parse_blocks(markdown_lines: &[Vec<Token>]) -> Vec<MdBlockElement> {
24    let mut block_elements: Vec<MdBlockElement> = Vec::new();
25
26    for line in markdown_lines {
27        if let Some(element) = parse_block(line) {
28            block_elements.push(element)
29        }
30    }
31
32    block_elements
33}

35/// Parses a single line of tokens into a block-level Markdown element.
36///
37/// # Arguments
38/// * `line` - A vector of tokens representing a single line of markdown.
39///
40/// # Returns
41/// An `Option<MdBlockElement>`, returning `None` for empty lines
42fn parse_block(line: &[Token]) -> Option<MdBlockElement> {
43    let first_token = line.first();
44
45    match first_token {
46        Some(Token::Punctuation(string)) if string == "#" => Some(parse_heading(line)),
47        Some(Token::Punctuation(string)) if string == "-" => {
48            // Note that setext headings have already been handled in the group_lines_to_blocks
49            // function by this point
50            if line.len() == 1 {
51                // If the line only contains a dash, then it is a thematic break
52                Some(MdBlockElement::ThematicBreak)
53            } else {
54                Some(parse_unordered_list(line))
55            }
56        }
57        Some(Token::OrderedListMarker(_)) => Some(parse_ordered_list(line)),
58        Some(Token::CodeFence) => Some(parse_codeblock(line)),
59        Some(Token::ThematicBreak) => Some(MdBlockElement::ThematicBreak),
60        Some(Token::TableCellSeparator) => Some(parse_table(line)),
61        Some(Token::BlockQuoteMarker) => Some(parse_blockquote(line)),
62        Some(Token::RawHtmlTag(_)) => Some(parse_raw_html(line)),
63        Some(Token::Tab) => Some(parse_indented_codeblock(line)),
64        Some(Token::Newline) => None,
65        _ => Some(MdBlockElement::Paragraph {
66            content: parse_inline(line),
67        }),
68    }
69}

71/// Parses an indented code block from a vector of tokens.
72///
73/// Note that CommonMark defines indented code blocks as lines that start with at least 4 spaces or
74/// a tab. However, this implementation only focuses on tabs, the size of which is defined in
75/// `config.toml`.
76///
77/// # Arguments
78/// * `line` - A vector of tokens representing an indented code block.
79///
80/// # Returns
81/// An `MdBlockElement::CodeBlock` containing the parsed code content.
82fn parse_indented_codeblock(line: &[Token]) -> MdBlockElement {
83    let mut code_content: Vec<String> = Vec::new();
84    let mut line_buffer: String = String::new();
85
86    let lines_split_by_newline = line
87        .split(|token| token == &Token::Newline)
88        .collect::<Vec<_>>();
89
90    lines_split_by_newline.iter().for_each(|token_line| {
91        if token_line.is_empty() {
92            return;
93        }
94
95        for token in &token_line[1..] {
96            match token {
97                Token::Tab => {
98                    line_buffer.push_str(&" ".repeat(CONFIG.get().unwrap().lexer.tab_size));
99                }
100                Token::Text(string) | Token::Punctuation(string) => line_buffer.push_str(string),
101                Token::Whitespace => line_buffer.push(' '),
102                Token::Newline => {
103                    push_buffer_to_collection(&mut code_content, &mut line_buffer);
104                    line_buffer.clear();
105                }
106                Token::Escape(esc_char) => {
107                    line_buffer.push_str(&format!("\\{esc_char}"));
108                }
109                Token::OrderedListMarker(string) => line_buffer.push_str(string),
110                Token::EmphasisRun { delimiter, length } => {
111                    line_buffer.push_str(&delimiter.to_string().repeat(*length))
112                }
113                Token::OpenParenthesis => line_buffer.push('('),
114                Token::CloseParenthesis => line_buffer.push(')'),
115                Token::OpenBracket => line_buffer.push('['),
116                Token::CloseBracket => line_buffer.push(']'),
117                Token::TableCellSeparator => line_buffer.push('|'),
118                Token::CodeTick => line_buffer.push('`'),
119                Token::CodeFence => line_buffer.push_str("```"),
120                Token::BlockQuoteMarker => line_buffer.push('>'),
121                Token::ThematicBreak => line_buffer.push_str("---"),
122                Token::RawHtmlTag(tag_content) => {
123                    // This should never be the first token, but inline html is allowed
124                    let escaped_tag = tag_content.replace("<", "&lt;").replace(">", "&gt;");
125                    line_buffer.push_str(&escaped_tag);
126                }
127            }
128        }
129
130        push_buffer_to_collection(&mut code_content, &mut line_buffer);
131    });
132
133    MdBlockElement::CodeBlock {
134        language: None,
135        lines: code_content,
136    }
137}

139/// Parses raw HTML tags from a vector of tokens into an `MdBlockElement::RawHtml`.
140///
141/// # Arguments
142/// * `line` - A vector of tokens representing a line of raw HTML.
143///
144/// # Returns
145/// An `MdBlockElement::RawHtml` containing the parsed HTML content.
146fn parse_raw_html(line: &[Token]) -> MdBlockElement {
147    let mut html_content = String::new();
148    for token in line {
149        match token {
150            Token::RawHtmlTag(tag_content) => html_content.push_str(tag_content),
151            Token::Text(string) | Token::Punctuation(string) => html_content.push_str(string),
152            Token::Whitespace => html_content.push(' '),
153            Token::Escape(esc_char) => {
154                html_content.push_str(&format!("\\{esc_char}"));
155            }
156            Token::Newline => html_content.push('\n'),
157            Token::OrderedListMarker(string) => html_content.push_str(string),
158            Token::EmphasisRun { delimiter, length } => {
159                html_content.push_str(&delimiter.to_string().repeat(*length))
160            }
161            Token::OpenParenthesis => html_content.push('('),
162            Token::CloseParenthesis => html_content.push(')'),
163            Token::OpenBracket => html_content.push('['),
164            Token::CloseBracket => html_content.push(']'),
165            Token::TableCellSeparator => html_content.push('|'),
166            Token::CodeTick => html_content.push('`'),
167            Token::CodeFence => html_content.push_str("```"),
168            Token::BlockQuoteMarker => html_content.push('>'),
169            Token::Tab => {
170                html_content.push_str(&" ".repeat(CONFIG.get().unwrap().lexer.tab_size));
171            }
172            Token::ThematicBreak => html_content.push_str("---"),
173        }
174    }
175
176    MdBlockElement::RawHtml {
177        content: html_content,
178    }

181/// Parses a blockquote from a vector of tokens into an `MdBlockElement::BlockQuote`.
182///
183/// # Arguments
184/// * `line` - A vector of tokens representing a blockquote.
185///
186/// # Returns
187/// An `MdBlockElement::BlockQuote` containing the parsed content, or a `MdBlockElement::Paragraph`
188/// if the content is empty.
189fn parse_blockquote(line: &[Token]) -> MdBlockElement {
190    let lines_split_by_newline = line
191        .split(|token| token == &Token::Newline)
192        .collect::<Vec<_>>();
193
194    let inner_blocks: Vec<Vec<Token>> = lines_split_by_newline
195        .iter()
196        .map(|tokens| {
197            let mut result = Vec::new();
198            if tokens.first() == Some(&Token::BlockQuoteMarker)
199                && tokens.get(1) == Some(&Token::Whitespace)
200            {
201                result.extend_from_slice(&tokens[2..]);
202            } else if tokens.first() == Some(&Token::BlockQuoteMarker) {
203                result.extend_from_slice(&tokens[1..]);
204            } else {
205                result.extend_from_slice(tokens);
206            }
207            result
208        })
209        .collect();
210
211    let grouped_inner_blocks = group_lines_to_blocks(inner_blocks);
212
213    let content = parse_blocks(&grouped_inner_blocks);
214
215    if content.is_empty() {
216        MdBlockElement::Paragraph {
217            content: parse_inline(line),
218        }
219    } else {
220        MdBlockElement::BlockQuote { content }
221    }
222}

224/// Parses a vector of tokens representing an ordered list into an `MdBlockElement::OrderedList`.
225///
226/// Calls the more generic `parse_list` function, which parses nested list items
227///
228/// # Arguments
229/// * `list` - A vector of tokens representing an ordered list.
230///
231/// # Returns
232/// An `MdBlockElement` representing the ordered list.
233fn parse_ordered_list(list: &[Token]) -> MdBlockElement {
234    parse_list(
235        list,
236        |tokens| {
237            matches!(
238                tokens.first(),
239                Some(Token::OrderedListMarker(_)) if tokens.get(1) == Some(&Token::Whitespace)
240            )
241        },
242        |items| MdBlockElement::OrderedList { items },
243    )
244}

246/// Parses a vector of tokens representing an unordered list into an `MdBlockElement::UnorderedList`.
247///
248/// Calls the more generic `parse_list` function, which parses nested list items
249///
250/// # Arguments
251/// * `list` - A vector of tokens representing an unordered list.
252///
253/// # Returns
254/// An `MdBlockElement` representing the unordered list.
255fn parse_unordered_list(list: &[Token]) -> MdBlockElement {
256    parse_list(
257        list,
258        |tokens| {
259            matches!(tokens.first(), Some(Token::Punctuation(string)) if string == "-" && tokens.get(1) == Some(&Token::Whitespace)
260            )
261        },
262        |items| MdBlockElement::UnorderedList { items },
263    )
264}

266/// Generic list parser used to reduce code duplication between ordered and unordered lists.
267///
268/// Handles splitting lines, identifying list items, and parsing nested lists. The behavior is
269/// determined by a predicate for identifying list items and a constructor for the resulting block.
270///
271/// # Arguments
272/// * `list` - The tokens to parse.
273/// * `is_list_item` - Predicate to identify a top-level list item.
274/// * `make_block` - Constructor for the resulting `MdBlockElement`.
275///
276/// # Returns
277/// An `MdBlockElement` representing either an ordered or unordered list, depending on the passed in constructor.
278fn parse_list<F, G>(list: &[Token], is_list_item: F, make_block: G) -> MdBlockElement
279where
280    F: Fn(&[Token]) -> bool,
281    G: Fn(Vec<MdListItem>) -> MdBlockElement,
282{
283    let lists_split_by_newline = list
284        .split(|token| token == &Token::Newline)
285        .collect::<Vec<_>>();
286    let mut list_items: Vec<MdListItem> = Vec::new();
287
288    let mut i = 0;
289    while i < lists_split_by_newline.len() {
290        let line = lists_split_by_newline[i];
291        if is_list_item(line) {
292            let content_tokens = line[2..].to_vec();
293            if let Some(content) = parse_block(&content_tokens) {
294                list_items.push(MdListItem { content })
295            }
296
297            // Check for consecutive tab-indented lines (nested list)
298            let mut nested_lines: Vec<Vec<Token>> = Vec::new();
299            let mut j = i + 1;
300            while j < lists_split_by_newline.len() {
301                let nested_line = lists_split_by_newline[j];
302                if nested_line.first() == Some(&Token::Tab) {
303                    let mut nested = nested_line.to_vec();
304                    while !nested.is_empty() && nested[0] == Token::Tab {
305                        nested.remove(0);
306                    }
307                    nested_lines.push(nested);
308                    j += 1;
309                } else {
310                    break;
311                }
312            }
313
314            if !nested_lines.is_empty() {
315                // Flatten nested lines into a single Vec<Token> separated by Newline
316                let mut nested_tokens: Vec<Token> = Vec::new();
317                for (k, l) in nested_lines.into_iter().enumerate() {
318                    if k > 0 {
319                        nested_tokens.push(Token::Newline);
320                    }
321                    nested_tokens.extend(l);
322                }
323
324                // Recursively parse nested list, try ordered first, fallback to unordered
325                let nested_block = if let Some(Token::OrderedListMarker(_)) = nested_tokens.first()
326                {
327                    parse_ordered_list(&nested_tokens)
328                } else {
329                    parse_unordered_list(&nested_tokens)
330                };
331
332                list_items.push(MdListItem {
333                    content: nested_block,
334                });
335
336                i = j - 1; // Skip processed nested lines
337            }
338        }
339        i += 1;
340    }
341
342    // Use the passed in constructor to create the List element
343    make_block(list_items)
344}

346/// Parses a vector of tokens representing a code block into an `MdBlockElement::CodeBlock`.
347///
348/// Extracts the language (if specified) and the code content.
349///
350/// # Arguments
351/// * `line` - A vector of tokens representing a code block.
352///
353/// # Returns
354/// An `MdBlockElement` representing the code block.
355fn parse_codeblock(line: &[Token]) -> MdBlockElement {
356    let mut code_content: Vec<String> = Vec::new();
357    let mut language = None;
358    let mut line_buffer: String = String::new();
359    let mut lines_split_by_newline = line
360        .split(|token| token == &Token::Newline)
361        .collect::<Vec<_>>();
362
363    if let Some(Token::Text(string)) = line.get(1) {
364        language = Some(string.clone());
365        lines_split_by_newline.remove(0);
366    }
367
368    lines_split_by_newline.iter().for_each(|line| {
369        if line.is_empty() {
370            return;
371        }
372
373        for token in line.iter() {
374            match token {
375                Token::Text(string) | Token::Punctuation(string) => line_buffer.push_str(string),
376                Token::Whitespace => line_buffer.push(' '),
377                Token::Newline => {
378                    push_buffer_to_collection(&mut code_content, &mut line_buffer);
379                    line_buffer.clear();
380                }
381                Token::Tab => {
382                    line_buffer.push_str(&" ".repeat(CONFIG.get().unwrap().lexer.tab_size));
383                }
384                Token::Escape(esc_char) => {
385                    line_buffer.push_str(&format!("\\{esc_char}"));
386                }
387                Token::OrderedListMarker(string) => line_buffer.push_str(string),
388                Token::EmphasisRun { delimiter, length } => {
389                    line_buffer.push_str(&delimiter.to_string().repeat(*length))
390                }
391                Token::OpenParenthesis => line_buffer.push('('),
392                Token::CloseParenthesis => line_buffer.push(')'),
393                Token::OpenBracket => line_buffer.push('['),
394                Token::CloseBracket => line_buffer.push(']'),
395                Token::TableCellSeparator => line_buffer.push('|'),
396                Token::CodeTick => line_buffer.push('`'),
397                Token::CodeFence => {}
398                Token::BlockQuoteMarker => line_buffer.push('>'),
399                Token::RawHtmlTag(tag_content) => {
400                    let escaped_tag = tag_content.replace("<", "&lt;").replace(">", "&gt;");
401                    line_buffer.push_str(&escaped_tag);
402                }
403                Token::ThematicBreak => line_buffer.push_str("---"),
404            }
405        }
406
407        push_buffer_to_collection(&mut code_content, &mut line_buffer);
408    });
409
410    push_buffer_to_collection(&mut code_content, &mut line_buffer);
411
412    MdBlockElement::CodeBlock {
413        language,
414        lines: code_content,
415    }
416}

418/// Parses a vector of tokens representing a heading into an `MdBlockElement::Header`.
419///
420/// Determines the heading level and parses the heading content.
421///
422/// # Arguments
423/// * `line` - A vector of tokens representing a heading line.
424///
425/// # Returns
426/// An `MdBlockElement` representing the heading, or a paragraph if the heading is invalid.
427fn parse_heading(line: &[Token]) -> MdBlockElement {
428    let mut heading_level = 0;
429    let mut i = 0;
430    while let Some(token) = line.get(i) {
431        match token {
432            Token::Punctuation(string) => {
433                if string == "#" {
434                    heading_level += 1;
435                } else {
436                    break;
437                }
438            }
439            _ => break,
440        }
441        i += 1;
442    }
443
444    // At this point, we should be at a non-# token or the end of the line
445    if i >= line.len() || line.get(i) != Some(&Token::Whitespace) {
446        return MdBlockElement::Paragraph {
447            content: parse_inline(line),
448        };
449    }
450
451    MdBlockElement::Header {
452        level: heading_level,
453        content: parse_inline(&line[i + 1..]),
454    }
455}

457/// Parses GitHub-style tables from the input vector of tokens.
458pub fn parse_table(line: &[Token]) -> MdBlockElement {
459    let rows = line
460        .split(|token| token == &Token::Newline)
461        .collect::<Vec<_>>();
462
463    if rows.len() < 3 {
464        return MdBlockElement::Paragraph {
465            content: parse_inline(line),
466        };
467    }
468
469    let header_row = rows
470        .first()
471        .expect("Table should have at least a header row")
472        .to_vec();
473
474    let alignment_row = rows
475        .get(1)
476        .expect("Table should have an alignment row")
477        .to_vec();
478
479    let alignments: Vec<TableAlignment> = split_row(&alignment_row)
480        .into_iter()
481        .map(|cell_content| {
482            let content: String = cell_content
483                .iter()
484                .filter_map(|token| match token {
485                    Token::Text(s) => {
486                        warn!("Table alignment should not contain text as it could result in unexpected behavior: {s}");
487                        Some(s.to_owned())
488                    }
489                    Token::Punctuation(s) => Some(s.to_owned()),
490                    Token::ThematicBreak => Some("---".to_string()),
491                    _ => None,
492                })
493                .collect();
494
495            match (content.starts_with(':'), content.ends_with(':')) {
496                (true, true) => TableAlignment::Center,
497                (true, false) => TableAlignment::Left,
498                (false, true) => TableAlignment::Right,
499                _ => TableAlignment::None,
500            }
501        })
502        .collect();
503
504    let headers: Vec<MdTableCell> = split_row(&header_row)
505        .into_iter()
506        .enumerate()
507        .map(|(i, cell_content)| MdTableCell {
508            content: parse_inline(cell_content),
509            alignment: alignments.get(i).cloned().unwrap_or(TableAlignment::None),
510            is_header: true,
511        })
512        .collect();
513
514    let body: Vec<Vec<MdTableCell>> = rows
515        .iter()
516        .skip(2)
517        .map(|row| {
518            split_row(row)
519                .into_iter()
520                .enumerate()
521                .map(|(i, cell_tokens)| MdTableCell {
522                    content: parse_inline(cell_tokens),
523                    alignment: alignments.get(i).cloned().unwrap_or(TableAlignment::None),
524                    is_header: false,
525                })
526                .collect()
527        })
528        .collect();
529
530    MdBlockElement::Table { headers, body }
531}

533/// Helper function to split a row of tokens into individual cells.
534///
535/// By removing the starting and ending "|" characters, it ensures that the row is
536/// split into the proper number of cells.
537fn split_row(row: &[Token]) -> Vec<&[Token]> {
538    let mut cells: Vec<&[Token]> = row
539        .split(|token| token == &Token::TableCellSeparator)
540        .collect();
541
542    if let Some(first) = cells.first() {
543        if first.is_empty() {
544            cells.remove(0);
545        }
546    }
547    if let Some(last) = cells.last() {
548        if last.is_empty() {
549            cells.pop();
550        }
551    }
552
553    cells
554}

556/// Parses a vector of tokens into a vector of inline Markdown elements (i.e. links, images,
557/// bold/italics, etc.).
558///
559/// # Arguments
560/// * `markdown_tokens` - A vector of tokens representing inline markdown content.
561///
562/// # Returns
563/// A vector of parsed inline Markdown elements.
564pub fn parse_inline(markdown_tokens: &[Token]) -> Vec<MdInlineElement> {
565    let mut parsed_inline_elements: Vec<MdInlineElement> = Vec::new();
566
567    let mut cursor: TokenCursor = TokenCursor {
568        tokens: markdown_tokens.to_vec(),
569        current_position: 0,
570    };
571
572    let mut delimiter_stack: Vec<Delimiter> = Vec::new();
573
574    let mut buffer: String = String::new();
575
576    let mut current_token: Token;
577    while !cursor.is_at_eof() {
578        current_token = cursor
579            .current()
580            .expect("Token should be valid markdown")
581            .clone();
582
583        match current_token {
584            Token::EmphasisRun { delimiter, length } => {
585                push_buffer_to_collection(&mut parsed_inline_elements, &mut buffer);
586
587                delimiter_stack.push(Delimiter {
588                    run_length: length,
589                    ch: delimiter,
590                    token_position: cursor.position(),
591                    parsed_position: parsed_inline_elements.len(),
592                    active: true,
593                    can_open: true,
594                    can_close: true,
595                });
596
597                parsed_inline_elements.push(MdInlineElement::Placeholder);
598            }
599            Token::OpenBracket => {
600                push_buffer_to_collection(&mut parsed_inline_elements, &mut buffer);
601
602                let link_element =
603                    parse_link_type(&mut cursor, |label, title, url| MdInlineElement::Link {
604                        text: label,
605                        title,
606                        url,
607                    });
608                parsed_inline_elements.push(link_element);
609            }
610            Token::CodeTick => {
611                // Search for a matching code tick, everything else is text
612                cursor.advance();
613                push_buffer_to_collection(&mut parsed_inline_elements, &mut buffer);
614
615                let code_content = parse_code_span(&mut cursor);
616
617                if cursor.current() != Some(&Token::CodeTick) {
618                    parsed_inline_elements.push(MdInlineElement::Text {
619                        content: format!("`{code_content}`"),
620                    });
621                } else {
622                    parsed_inline_elements.push(MdInlineElement::Code {
623                        content: code_content,
624                    });
625                }
626            }
627            Token::Punctuation(string) if string == "!" => {
628                if cursor.peek_ahead(1) != Some(&Token::OpenBracket) {
629                    // If the next token is not an open bracket, treat it as text
630                    buffer.push('!');
631                    cursor.advance();
632                    continue;
633                }
634
635                push_buffer_to_collection(&mut parsed_inline_elements, &mut buffer);
636                cursor.advance(); // Advance to the open bracket
637
638                let image =
639                    parse_link_type(&mut cursor, |label, title, url| MdInlineElement::Image {
640                        alt_text: flatten_inline(&label),
641                        title,
642                        url,
643                    });
644
645                parsed_inline_elements.push(image);
646            }
647            Token::Escape(esc_char) => buffer.push_str(&format!("\\{esc_char}")),
648            Token::Text(string) | Token::Punctuation(string) => buffer.push_str(&string),
649            Token::OrderedListMarker(string) => buffer.push_str(&string),
650            Token::Whitespace => buffer.push(' '),
651            Token::CloseBracket => buffer.push(']'),
652            Token::OpenParenthesis => buffer.push('('),
653            Token::CloseParenthesis => buffer.push(')'),
654            Token::ThematicBreak => buffer.push_str("---"),
655            Token::TableCellSeparator => buffer.push('|'),
656            Token::BlockQuoteMarker => buffer.push('>'),
657            Token::RawHtmlTag(tag_content) => buffer.push_str(&tag_content),
658            _ => push_buffer_to_collection(&mut parsed_inline_elements, &mut buffer),
659        }
660
661        cursor.advance();
662    }
663
664    push_buffer_to_collection(&mut parsed_inline_elements, &mut buffer);
665
666    delimiter_stack
667        .iter_mut()
668        .for_each(|el| el.classify_flanking(&cursor.tokens));
669
670    resolve_emphasis(&mut parsed_inline_elements, &mut delimiter_stack);
671
672    parsed_inline_elements
673}

675/// Parses a code span starting from the current position of the cursor.
676///
677/// # Arguments
678/// * `cursor` - A mutable reference to a `TokenCursor` that tracks the current position in the
679///
680/// # Returns
681/// A string containing the content of the code span, excluding the opening and closing code ticks.
682fn parse_code_span(cursor: &mut TokenCursor) -> String {
683    let mut code_content: String = String::new();
684    while let Some(next_token) = cursor.current() {
685        match next_token {
686            Token::CodeTick => break,
687            Token::Text(string) | Token::Punctuation(string) => code_content.push_str(string),
688            Token::OrderedListMarker(string) => code_content.push_str(string),
689            Token::Escape(ch) => code_content.push_str(&format!("\\{ch}")),
690            Token::OpenParenthesis => code_content.push('('),
691            Token::CloseParenthesis => code_content.push(')'),
692            Token::OpenBracket => code_content.push('['),
693            Token::CloseBracket => code_content.push(']'),
694            Token::TableCellSeparator => code_content.push('|'),
695            Token::EmphasisRun { delimiter, length } => {
696                code_content.push_str(&delimiter.to_string().repeat(*length))
697            }
698            Token::Whitespace => code_content.push(' '),
699            Token::Tab => code_content.push_str(&" ".repeat(CONFIG.get().unwrap().lexer.tab_size)),
700            Token::Newline => code_content.push('\n'),
701            Token::ThematicBreak => code_content.push_str("---"),
702            Token::BlockQuoteMarker => code_content.push('>'),
703            Token::RawHtmlTag(tag_content) => code_content.push_str(tag_content),
704            Token::CodeFence => {}
705        }
706
707        cursor.advance();
708    }
709
710    code_content
711}

731/// Parses a link type (either a link or an image) from the current position of the cursor.
732///
733/// # Arguments
734/// * `cursor` - A mutable reference to a `TokenCursor` that tracks the current position in the
735///   token stream.
736/// * `make_element` - A closure that takes the parsed label elements, optional title, and URI,
737///   and returns an `MdInlineElement` representing the link or image.
738///
739/// # Returns
740/// An `MdInlineElement` representing the parsed link or image.
741fn parse_link_type<F>(cursor: &mut TokenCursor, make_element: F) -> MdInlineElement
742where
743    F: Fn(Vec<MdInlineElement>, Option<String>, String) -> MdInlineElement,
744{
745    let mut label_elements: Vec<MdInlineElement> = Vec::new();
746    let mut label_buffer = String::new();
747    let mut delimiter_stack: Vec<Delimiter> = Vec::new();
748    cursor.advance(); // Move past the open bracket
749    while let Some(token) = cursor.current() {
750        match token {
751            Token::CloseBracket => {
752                push_buffer_to_collection(&mut label_elements, &mut label_buffer);
753                break;
754            }
755            Token::OpenBracket => {
756                push_buffer_to_collection(&mut label_elements, &mut label_buffer);
757
758                let inner_link = parse_link_type(cursor, make_link);
759                label_elements.push(inner_link);
760            }
761            Token::EmphasisRun { delimiter, length } => {
762                push_buffer_to_collection(&mut label_elements, &mut label_buffer);
763                delimiter_stack.push(Delimiter {
764                    run_length: *length,
765                    ch: *delimiter,
766                    token_position: cursor.position(),
767                    parsed_position: label_elements.len(),
768                    active: true,
769                    can_open: true,
770                    can_close: true,
771                });
772                label_elements.push(MdInlineElement::Placeholder);
773            }
774            Token::Punctuation(s) if s == "!" => {
775                if cursor.peek_ahead(1) != Some(&Token::OpenBracket) {
776                    label_buffer.push('!');
777                    cursor.advance();
778                    continue;
779                }
780
781                push_buffer_to_collection(&mut label_elements, &mut label_buffer);
782                cursor.advance(); // Advance to the open bracket
783                let inner_image = parse_link_type(cursor, make_image);
784
785                label_elements.push(inner_image);
786            }
787            Token::Text(s) | Token::Punctuation(s) => label_buffer.push_str(s),
788            Token::OrderedListMarker(s) => label_buffer.push_str(s),
789            Token::Escape(ch) => label_buffer.push_str(&format!("\\{ch}")),
790            Token::Whitespace => label_buffer.push(' '),
791            Token::ThematicBreak => label_buffer.push_str("---"),
792            Token::OpenParenthesis => label_buffer.push('('),
793            Token::CloseParenthesis => label_buffer.push(')'),
794            Token::TableCellSeparator => label_buffer.push('|'),
795            Token::BlockQuoteMarker => label_buffer.push('>'),
796            _ => {}
797        }
798        cursor.advance();
799    }
800
801    push_buffer_to_collection(&mut label_elements, &mut label_buffer);
802    resolve_emphasis(&mut label_elements, &mut delimiter_stack);
803
804    // If we didn't find a closing bracket, treat it as text
805    if cursor.current() != Some(&Token::CloseBracket) {
806        return MdInlineElement::Text {
807            content: format!("[{}", flatten_inline(&label_elements)),
808        };
809    }
810
811    // At this point we should have parentheses for the uri, otherwise treat it as a
812    // text element
813    if cursor.peek_ahead(1) != Some(&Token::OpenParenthesis) {
814        cursor.advance();
815        return MdInlineElement::Text {
816            content: format!("[{}]", flatten_inline(&label_elements)),
817        };
818    }
819
820    cursor.advance(); // Move to '('
821
822    let mut uri = String::new();
823    let mut title = String::new();
824    let mut is_building_title = false;
825    let mut is_valid_title = true;
826    let mut has_opening_quote = false;
827
828    while let Some(token) = cursor.current() {
829        if !is_building_title {
830            match token {
831                Token::CloseParenthesis => break,
832                Token::Text(s) | Token::Punctuation(s) => uri.push_str(s),
833                Token::OrderedListMarker(s) => uri.push_str(s),
834                Token::Escape(ch) => uri.push_str(&format!("\\{ch}")),
835                Token::Whitespace => is_building_title = true,
836                Token::ThematicBreak => uri.push_str("---"),
837                Token::TableCellSeparator => uri.push('|'),
838                Token::BlockQuoteMarker => uri.push('>'),
839                Token::RawHtmlTag(tag_content) => uri.push_str(tag_content),
840                _ => {}
841            }
842        } else {
843            match token {
844                Token::CloseParenthesis => break,
845                Token::Punctuation(s) if s == "\"" => {
846                    if has_opening_quote {
847                        is_valid_title = true;
848                        is_building_title = false;
849                    } else {
850                        has_opening_quote = true;
851                        is_valid_title = false;
852                    }
853                }
854                Token::Text(s) | Token::Punctuation(s) => title.push_str(s),
855                Token::OrderedListMarker(s) => title.push_str(s),
856                Token::Escape(ch) => title.push_str(&format!("\\{ch}")),
857                Token::EmphasisRun { delimiter, length } => {
858                    title.push_str(&delimiter.to_string().repeat(*length))
859                }
860                Token::OpenBracket => title.push('['),
861                Token::CloseBracket => title.push(']'),
862                Token::OpenParenthesis => title.push('('),
863                Token::TableCellSeparator => title.push('|'),
864                Token::Tab => title.push('\t'),
865                Token::Newline => title.push_str("\\n"),
866                Token::Whitespace => title.push(' '),
867                Token::CodeTick => title.push('`'),
868                Token::CodeFence => title.push_str("```"),
869                Token::ThematicBreak => title.push_str("---"),
870                Token::BlockQuoteMarker => title.push('>'),
871                Token::RawHtmlTag(tag_content) => {
872                    warn!(
873                        "Raw HTML tags in titles can result in unexpected behavior: {tag_content}"
874                    );
875                    title.push_str(tag_content);
876                }
877            }
878        }
879        cursor.advance();
880    }
881
882    // If we didn't find a closing parenthesis or if the title is invalid, treat it as text
883    if cursor.current() != Some(&Token::CloseParenthesis) {
884        return MdInlineElement::Text {
885            content: format!("[{}]({} ", flatten_inline(&label_elements), uri),
886        };
887    } else if !title.is_empty() && !is_valid_title {
888        return MdInlineElement::Text {
889            content: format!("[{}]({} {})", flatten_inline(&label_elements), uri, title),
890        };
891    }
892
893    make_element(label_elements, Some(title).filter(|t| !t.is_empty()), uri)
894}

896/// Flattens a vector of inline Markdown elements into a single string.
897///
898/// # Arguments
899/// * `elements` - A vector of inline Markdown elements to flatten.
900///
901/// # Returns
902/// A string containing the concatenated content of all inline elements
903fn flatten_inline(elements: &[MdInlineElement]) -> String {
904    let mut result = String::new();
905    for element in elements {
906        match element {
907            MdInlineElement::Text { content } => result.push_str(content),
908            MdInlineElement::Bold { content } => result.push_str(&flatten_inline(content)),
909            MdInlineElement::Italic { content } => result.push_str(&flatten_inline(content)),
910            MdInlineElement::Code { content } => result.push_str(content),
911            MdInlineElement::Link { text, .. } => result.push_str(&flatten_inline(text)),
912            MdInlineElement::Image { alt_text, .. } => result.push_str(alt_text),
913            _ => {}
914        }
915    }
916    result
917}

919/// Parses (resolves) emphasis in a vector of inline Markdown elements.
920///
921/// Modifies the elements in place to convert delimiter runs into bold or italic elements as appropriate.
922///
923/// # Arguments
924/// * `elements` - A mutable reference to a vector of inline Markdown elements.
925/// * `delimiter_stack` - A mutable reference to a slice of delimiters.
926fn resolve_emphasis(elements: &mut Vec<MdInlineElement>, delimiter_stack: &mut [Delimiter]) {
927    if delimiter_stack.len() == 1 {
928        // If there is only one delimiter, it cannot be resolved to emphasis
929        if delimiter_stack[0].active {
930            elements[delimiter_stack[0].parsed_position] = MdInlineElement::Text {
931                content: delimiter_stack[0].ch.to_string(),
932            };
933        }
934        return;
935    }
936
937    for i in 0..delimiter_stack.len() {
938        if !delimiter_stack[i].active || !delimiter_stack[i].can_close {
939            continue;
940        }
941
942        // At this point we have a valid closer
943        let closer = delimiter_stack[i].clone();
944
945        for j in (0..i).rev() {
946            if !delimiter_stack[j].active || !delimiter_stack[j].can_open {
947                continue;
948            }
949
950            let opener = delimiter_stack[j].clone();
951
952            // Check if the opener and closer have the same delimiter
953            if !closer.ch.eq(&opener.ch) {
954                continue;
955            }
956
957            // Rule of 3: If the total length of the run is a multiple of 3 and both run lengths
958            // are not divisible by 3, they are not valid for emphasis
959            let length_total = closer.run_length + opener.run_length;
960            if ((closer.can_open && closer.can_close) || (opener.can_open && opener.can_close))
961                && (length_total % 3 == 0
962                    && closer.run_length % 3 != 0
963                    && opener.run_length % 3 != 0)
964            {
965                continue;
966            }
967
968            // Prefer making bold connections first
969            let delimiters_used = if closer.run_length >= 2 && opener.run_length >= 2 {
970                2
971            } else {
972                1
973            };
974
975            // Replace the placeholders with the new element
976            let range_start = if opener.run_length > delimiters_used {
977                opener.parsed_position + 1
978            } else {
979                opener.parsed_position
980            };
981
982            let range_end = if closer.run_length >= delimiters_used {
983                closer.parsed_position
984            } else {
985                closer.parsed_position + 1
986            };
987
988            // Map the delimiters used to bold/italic respectively
989            let element_to_insert = match delimiters_used {
990                2 => MdInlineElement::Bold {
991                    content: elements[range_start + 1..range_end].to_vec(),
992                },
993                1 => MdInlineElement::Italic {
994                    content: elements[range_start + 1..range_end].to_vec(),
995                },
996                _ => unreachable!(),
997            };
998
999            elements.splice(range_start..=range_end, vec![element_to_insert]);
1000            let num_elements_removed = range_end - range_start;
1001
1002            // closer.parsed_position -= num_elements_removed;
1003
1004            // Update the parsed positions of the delimiters
1005            (0..delimiter_stack.len()).for_each(|k| {
1006                if delimiter_stack[k].parsed_position > closer.parsed_position {
1007                    delimiter_stack[k].parsed_position -= num_elements_removed;
1008                }
1009            });
1010
1011            delimiter_stack[i].run_length = delimiter_stack[i]
1012                .run_length
1013                .saturating_sub(delimiters_used);
1014            delimiter_stack[j].run_length = delimiter_stack[j]
1015                .run_length
1016                .saturating_sub(delimiters_used);
1017
1018            if delimiter_stack[i].run_length == 0 {
1019                delimiter_stack[i].active = false;
1020            }
1021            if delimiter_stack[j].run_length == 0 {
1022                delimiter_stack[j].active = false;
1023            }
1024        }
1025    }
1026
1027    // For all delimiters that are still active, replace the placeholders with Text elements
1028    delimiter_stack.iter_mut().for_each(|el| {
1029        if el.active && el.parsed_position < elements.len() {
1030            elements[el.parsed_position] = MdInlineElement::Text {
1031                content: el.ch.to_string(),
1032            };
1033        }
1034    });
1035}

1037/// Groups adjacent tokenized lines into groups (blocks) for further parsing.
1038///
1039/// # Arguments
1040/// * `tokenized_lines` - A vector of vectors, where each inner vector contains tokens representing a line of markdown.
1041///
1042/// # Returns
1043/// A vector of vectors, where each inner vector represents a grouped block of tokens.
1044pub fn group_lines_to_blocks(mut tokenized_lines: Vec<Vec<Token>>) -> Vec<Vec<Token>> {
1045    let mut blocks: Vec<Vec<Token>> = Vec::new();
1046    let mut current_block: Vec<Token> = Vec::new();
1047    let mut previous_block: Vec<Token>;
1048    let lines = tokenized_lines.iter_mut();
1049    let mut is_inside_code_block = false;
1050    for line in lines {
1051        previous_block = blocks.last().unwrap_or(&Vec::new()).to_vec();
1052
1053        // Appending all tokens between two code fences to one block
1054        if is_inside_code_block && line.first() != Some(&Token::CodeFence) {
1055            // If we are inside a code block, then we just append the line to the current block
1056            attach_to_previous_block(&mut blocks, &mut previous_block, line, Some(Token::Newline));
1057            continue;
1058        } else if is_inside_code_block && line.first() == Some(&Token::CodeFence) {
1059            // If we are inside a code block and the line starts with a code fence, then we end the
1060            // code block
1061            is_inside_code_block = false;
1062            attach_to_previous_block(&mut blocks, &mut previous_block, line, None);
1063            continue;
1064        }
1065
1066        match line.first() {
1067            Some(Token::Punctuation(string)) if string == "#" => {
1068                // For ATX headings, it must all be on one line
1069                blocks.push(line.to_owned());
1070            }
1071            Some(Token::Punctuation(string)) if string == "-" => {
1072                group_dashed_lines(&mut blocks, &mut current_block, &mut previous_block, line);
1073            }
1074            Some(Token::Tab) => {
1075                group_tabbed_lines(&mut blocks, &mut current_block, &mut previous_block, line);
1076            }
1077            Some(Token::OrderedListMarker(_)) => {
1078                group_ordered_list(&mut blocks, &mut current_block, &mut previous_block, line);
1079            }
1080            Some(Token::ThematicBreak) => {
1081                // Check if the previous line starts with anything other than a heading
1082                // If so, then this is actually a setext heading 2
1083                if let Some(previous_line_start) = previous_block.first() {
1084                    match previous_line_start {
1085                        Token::Punctuation(string) if string == "#" => {
1086                            blocks.push(line.to_owned());
1087                        }
1088                        Token::Newline => blocks.push(line.to_owned()),
1089                        _ => {
1090                            previous_block.insert(0, Token::Punctuation(String::from("#")));
1091                            previous_block.insert(1, Token::Punctuation(String::from("#")));
1092                            previous_block.insert(2, Token::Whitespace);
1093                            blocks.pop();
1094                            blocks.push(previous_block.clone());
1095                        }
1096                    }
1097                } else {
1098                    current_block.extend(line.to_owned());
1099                }
1100            }
1101            Some(Token::BlockQuoteMarker) => {
1102                if let Some(previous_line_start) = previous_block.first() {
1103                    if matches!(previous_line_start, Token::BlockQuoteMarker) {
1104                        attach_to_previous_block(
1105                            &mut blocks,
1106                            &mut previous_block,
1107                            line,
1108                            Some(Token::Newline),
1109                        );
1110                    } else {
1111                        current_block.extend(line.to_owned());
1112                    }
1113                } else {
1114                    current_block.extend(line.to_owned());
1115                }
1116            }
1117            Some(Token::CodeTick) => {
1118                current_block.extend(line.to_owned());
1119            }
1120            Some(Token::CodeFence) => {
1121                if !is_inside_code_block {
1122                    is_inside_code_block = true;
1123                    current_block.extend(line.to_owned());
1124                } else {
1125                    is_inside_code_block = false;
1126                    current_block.extend(line.to_owned());
1127                    blocks.push(current_block.clone());
1128                    current_block.clear();
1129                }
1130            }
1131            Some(Token::Text(string)) if string == "=" => {
1132                let has_trailing_content = line.iter().skip(1).any(|token| match token {
1133                    Token::Text(s) if s == "=" => false,
1134                    Token::Whitespace | Token::Tab | Token::Newline => false,
1135                    _ => true,
1136                });
1137
1138                // Setext heading 1
1139                if let Some(previous_line_start) = previous_block.first() {
1140                    if !has_trailing_content && matches!(previous_line_start, Token::Text(_)) {
1141                        group_setext_heading_one(&mut blocks, &mut previous_block);
1142                    } else {
1143                        group_text_lines(
1144                            &mut blocks,
1145                            &mut current_block,
1146                            &mut previous_block,
1147                            line,
1148                        );
1149                    }
1150                } else {
1151                    current_block.extend(line.to_owned());
1152                }
1153            }
1154            Some(Token::Text(_)) => {
1155                group_text_lines(&mut blocks, &mut current_block, &mut previous_block, line);
1156            }
1157            Some(Token::TableCellSeparator) => {
1158                group_table_rows(&mut blocks, &mut current_block, &mut previous_block, line);
1159            }
1160            Some(Token::Whitespace) => {
1161                group_lines_with_leading_whitespace(
1162                    &mut blocks,
1163                    &mut current_block,
1164                    &mut previous_block,
1165                    line,
1166                );
1167            }
1168            _ => {
1169                // Catch-all for everything else
1170                current_block.extend(line.to_owned());
1171            }
1172        }
1173
1174        if !current_block.is_empty() {
1175            blocks.push(current_block.clone());
1176        }
1177
1178        current_block.clear();
1179    }
1180    blocks
1181}

1208/// Groups text lines into blocks based on the previous block's content.
1209///
1210/// # Arguments
1211/// * `blocks` - A mutable reference to a vector of blocks, where each block is a vector of tokens.
1212/// * `current_block` - A mutable reference to the current block being processed.
1213/// * `previous_block` - A mutable reference to the previous block, used for context.
1214/// * `line` - A mutable reference to the current line being processed, which is a vector of
1215///   tokens.
1216fn group_text_lines(
1217    blocks: &mut Vec<Vec<Token>>,
1218    current_block: &mut Vec<Token>,
1219    previous_block: &mut Vec<Token>,
1220    line: &[Token],
1221) {
1222    if !previous_block.is_empty() {
1223        if matches!(previous_block.first(), Some(Token::Text(_))) {
1224            attach_to_previous_block(blocks, previous_block, line, Some(Token::Whitespace));
1225        } else if matches!(previous_block.first(), Some(Token::Punctuation(_))) {
1226            // If the previous block was a heading, then this is a new paragraph
1227            current_block.extend(line.to_owned());
1228        } else {
1229            // If the previous block was empty, then this is a new paragraph
1230            current_block.extend(line.to_owned());
1231        }
1232    } else {
1233        // If the previous block was empty, then this is a new paragraph
1234        current_block.extend(line.to_owned());
1235    }
1236}

1253/// Groups ordered list lines into a block by appending the line to the previous block if it is
1254/// part of the same list.
1255///
1256/// # Arguments
1257/// * `blocks` - A mutable reference to a vector of blocks, where each block is a vector of tokens.
1258/// * `current_block` - A mutable reference to the current block being processed.
1259/// * `previous_block` - A mutable reference to the previous block, used for context.
1260/// * `line` - A mutable reference to the current line being processed, which is a vector of
1261///   tokens.
1262fn group_ordered_list(
1263    blocks: &mut Vec<Vec<Token>>,
1264    current_block: &mut Vec<Token>,
1265    previous_block: &mut Vec<Token>,
1266    line: &[Token],
1267) {
1268    if let Some(previous_line_start) = previous_block.first() {
1269        match previous_line_start {
1270            Token::OrderedListMarker(_) if previous_block.get(1) == Some(&Token::Whitespace) => {
1271                // If the previous block is a list, then we append the line to it
1272                attach_to_previous_block(blocks, previous_block, line, Some(Token::Newline));
1273            }
1274            _ => {
1275                current_block.extend(line.to_owned());
1276            }
1277        }
1278    } else {
1279        current_block.extend(line.to_owned());
1280    }
1281}

1283/// Attaches the current line to the previous block, optionally adding a separator token.
1284fn attach_to_previous_block(
1285    blocks: &mut Vec<Vec<Token>>,
1286    previous_block: &mut Vec<Token>,
1287    line: &[Token],
1288    separator: Option<Token>,
1289) {
1290    if let Some(separator) = separator {
1291        previous_block.push(separator);
1292    }
1293
1294    previous_block.extend(line.to_owned());
1295    blocks.pop();
1296    blocks.push(previous_block.clone());
1297}

1299/// Groups tabbed lines into blocks based on the previous block's content.
1300///
1301/// Note that this function short-circuits when the first token of the line is a raw HTML tag,
1302/// to allow for indented HTML.
1303///
1304/// # Arguments
1305/// * `blocks` - A mutable reference to a vector of blocks, where each block is a vector of tokens.
1306/// * `current_block` - A mutable reference to the current block being processed.
1307/// * `previous_block` - A mutable reference to the previous block, used for context.
1308/// * `line` - A mutable reference to the current line being processed, which is a vector of
1309///   tokens.
1310fn group_tabbed_lines(
1311    blocks: &mut Vec<Vec<Token>>,
1312    current_block: &mut Vec<Token>,
1313    previous_block: &mut Vec<Token>,
1314    line: &[Token],
1315) {
1316    if line.len() == 1 {
1317        current_block.extend(line.to_owned());
1318        return;
1319    }
1320
1321    let non_whitespace_index = line
1322        .iter()
1323        .position(|token| !matches!(token, Token::Whitespace | Token::Tab | Token::Newline));
1324
1325    if let Some(first_content_token) = line.get(non_whitespace_index.unwrap_or(0)) {
1326        if matches!(first_content_token, Token::RawHtmlTag(_))
1327            && matches!(previous_block.first(), Some(Token::RawHtmlTag(_)))
1328        {
1329            // If the first token is a raw HTML tag, we attach the line to the previous block
1330            let line_to_attach = line
1331                .iter()
1332                .skip_while(|t| matches!(t, Token::Whitespace | Token::Tab | Token::Newline))
1333                .cloned()
1334                .collect::<Vec<Token>>();
1335
1336            attach_to_previous_block(
1337                blocks,
1338                previous_block,
1339                &line_to_attach,
1340                Some(Token::Newline),
1341            );
1342
1343            return;
1344        } else if matches!(first_content_token, Token::RawHtmlTag(_)) {
1345            current_block.extend(
1346                line.iter()
1347                    .skip_while(|t| matches!(t, Token::Whitespace | Token::Tab | Token::Newline))
1348                    .cloned(),
1349            );
1350            return;
1351        }
1352
1353        if !previous_block.is_empty() {
1354            let previous_line_start = previous_block.first();
1355            match previous_line_start {
1356                Some(Token::Punctuation(string))
1357                    if string == "-" && previous_block.get(1) == Some(&Token::Whitespace) =>
1358                {
1359                    // If the previous block is a list, then we append the line to it
1360                    attach_to_previous_block(blocks, previous_block, line, Some(Token::Newline));
1361                }
1362                Some(Token::OrderedListMarker(_))
1363                    if previous_block.get(1) == Some(&Token::Whitespace) =>
1364                {
1365                    // If the previous block is an ordered list, then we append the
1366                    // line to it
1367                    attach_to_previous_block(blocks, previous_block, line, Some(Token::Newline));
1368                }
1369                Some(Token::RawHtmlTag(_)) => {
1370                    attach_to_previous_block(blocks, previous_block, line, Some(Token::Newline));
1371                }
1372                Some(Token::Tab) => {
1373                    attach_to_previous_block(blocks, previous_block, line, Some(Token::Newline));
1374                }
1375                _ => {
1376                    // If the previous block is not a list, then we just add the
1377                    // line to the current block
1378                    current_block.extend(line.to_owned());
1379                }
1380            }
1381        } else {
1382            // If the previous block is empty, then we just add the line to the
1383            // current block
1384            current_block.extend(line.to_owned());
1385        }
1386    }
1387}

1389/// Groups lines with leading whitespace into blocks based on the previous block's content.
1390///
1391/// # Arguments
1392/// * `blocks` - A mutable reference to a vector of blocks, where each block is a vector of tokens.
1393/// * `current_block` - A mutable reference to the current block being processed.
1394/// * `previous_block` - A mutable reference to the previous block, used for context.
1395/// * `line` - A mutable reference to the current line being processed, which is a vector of
1396///   tokens.
1397fn group_lines_with_leading_whitespace(
1398    blocks: &mut Vec<Vec<Token>>,
1399    current_block: &mut Vec<Token>,
1400    previous_block: &mut Vec<Token>,
1401    line: &[Token],
1402) {
1403    if let Some(first_content_token) = line
1404        .iter()
1405        .find(|t| !matches!(t, Token::Whitespace | Token::Tab | Token::Newline))
1406    {
1407        if let Some(previous_line_start) = previous_block.first() {
1408            match previous_line_start {
1409                Token::Whitespace => {
1410                    // Check if the previous line has non-whitespace content
1411                    if line
1412                        .iter()
1413                        .any(|t| !matches!(t, Token::Whitespace | Token::Tab | Token::Newline))
1414                    {
1415                        attach_to_previous_block(
1416                            blocks,
1417                            previous_block,
1418                            line,
1419                            Some(Token::Newline),
1420                        );
1421                    } else {
1422                        current_block.extend(line.to_owned());
1423                    }
1424                }
1425                Token::RawHtmlTag(_) => {
1426                    if matches!(first_content_token, Token::RawHtmlTag(_)) {
1427                        // If the first token is a raw HTML tag, we attach the line to the previous block
1428                        attach_to_previous_block(
1429                            blocks,
1430                            previous_block,
1431                            line,
1432                            Some(Token::Newline),
1433                        );
1434                    } else {
1435                        current_block.extend(line.to_owned());
1436                    }
1437                }
1438                Token::Punctuation(string) if string == "-" => {
1439                    if matches!(first_content_token, Token::Punctuation(_)) {
1440                        attach_to_previous_block(
1441                            blocks,
1442                            previous_block,
1443                            line,
1444                            Some(Token::Newline),
1445                        );
1446                    } else {
1447                        current_block.extend(line.to_owned());
1448                    }
1449                }
1450                Token::Text(_) | Token::Punctuation(_) => {
1451                    attach_to_previous_block(blocks, previous_block, line, Some(Token::Newline));
1452                }
1453                _ => {
1454                    // Append the line to current block, excluding leading whitespace
1455                    current_block.extend(
1456                        line.iter()
1457                            .skip_while(|t| {
1458                                matches!(t, Token::Whitespace | Token::Tab | Token::Newline)
1459                            })
1460                            .cloned(),
1461                    );
1462                }
1463            }
1464        } else {
1465            current_block.extend(line.to_owned());
1466        }
1467    }
1468}

1470/// Groups dashed lines into blocks based on the previous block's content.
1471///
1472/// # Arguments
1473/// * `blocks` - A mutable reference to a vector of blocks, where each block is a vector of tokens.
1474/// * `current_block` - A mutable reference to the current block being processed.
1475/// * `previous_block` - A mutable reference to the previous block, used for context.
1476/// * `line` - A mutable reference to the current line being processed, which is a vector of
1477///   tokens.
1478fn group_dashed_lines(
1479    blocks: &mut Vec<Vec<Token>>,
1480    current_block: &mut Vec<Token>,
1481    previous_block: &mut Vec<Token>,
1482    line: &[Token],
1483) {
1484    if let Some(previous_line_start) = previous_block.first() {
1485        match previous_line_start {
1486            Token::Punctuation(string)
1487                if string == "-" && previous_block.get(1) == Some(&Token::Whitespace) =>
1488            {
1489                // Then it is either the start of a list or part of a list
1490
1491                attach_to_previous_block(blocks, previous_block, line, Some(Token::Newline));
1492            }
1493            Token::Punctuation(string) if string == "#" => {
1494                blocks.push(line.to_owned());
1495            }
1496            _ => {
1497                if line.len() > 1 {
1498                    current_block.extend(line.to_owned());
1499                } else {
1500                    // Then this is a Setext heading 2
1501                    previous_block.insert(0, Token::Punctuation(String::from("#")));
1502                    previous_block.insert(1, Token::Punctuation(String::from("#")));
1503                    previous_block.insert(2, Token::Whitespace);
1504                    blocks.pop();
1505                    blocks.push(previous_block.clone());
1506                }
1507            }
1508        }
1509    } else {
1510        current_block.extend(line.to_owned());
1511    }
1512}
